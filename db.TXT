CREATE TABLE AccessControlCardRec(
id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
EventId INT(11) UNSIGNED,
DeviceId INT(11) UNSIGNED,
CardName VARCHAR(30),
CardNo VARCHAR(30),
CreateTime INT(11) UNSIGNED,
Door INT(2) UNSIGNED,
Method INT(2) UNSIGNED,
Notes VARCHAR(30),
Password VARCHAR(50),
RecNo INT(11) UNSIGNED,
ReservedInt INT(2) UNSIGNED,
ReservedString VARCHAR(10),
RoomNumber VARCHAR(30),
Status INT(2) UNSIGNED,
URL VARCHAR(255),
UserID VARCHAR(30)
);


CREATE TABLE VideoTalkLog(
id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
EventId INT(11) UNSIGNED,
DeviceId INT(11) UNSIGNED,
CallType VARCHAR(30),
CreateTime INT(11) UNSIGNED,
EndState VARCHAR(30),
LocalNumber VARCHAR(30),
MessageTime INT(5) UNSIGNED,
Notes VARCHAR(30),
PeerNumber VARCHAR(30),
PeerType VARCHAR(10),
PicturePath VARCHAR(250),
RecNo INT(11) UNSIGNED,
Reserved VARCHAR(10),
TalkTime INT(5) UNSIGNED
);



CREATE TABLE AlarmRecord(
id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
EventId INT(11) UNSIGNED,
DeviceId INT(11) UNSIGNED,
Channel INT(2) UNSIGNED,
CreateTime INT(11) UNSIGNED,
Notes VARCHAR(30),
ReadFlag INT(1) UNSIGNED,
RecNo INT(11) UNSIGNED,
ReservedInt INT(2) UNSIGNED,
ReservedString VARCHAR(10),
RoomNumber VARCHAR(30),
SenseMethod VARCHAR(30)
);
